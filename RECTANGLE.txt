#include<stdio.h>

#include<math.h>

int main()

{

    int length,width,perimeter,area,diagonal;

    printf("enter the length and width\n");

    scanf("%d","%d", &length, &width);

    area=length*width;

    perimeter=2*(length+width);

    diagonal=sqrt((length*length)+(width*width));

    printf("The area of rectangle= %d\n", area);

    printf("The perimeter of rectangle= %d\n", perimeter);

    printf("The length of diagonal of rectangle= %d\n", diagonal);

    return 0;

    

}
